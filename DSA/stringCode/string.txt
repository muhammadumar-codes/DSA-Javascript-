 Common String Escape Characters :


| Escape | Meaning      | Example              |
| ------ | ------------ | -------------------- |
| `\'`   | Single quote | `'It\'s fine'`       |
| `\"`   | Double quote | `"He said \"Hi\""`   |
| `\\`   | Backslash    | `"C:\\Users\\Janii"` |
| `\n`   | New line     | `"Hello\nWorld"`     |
| `\t`   | Tab          | `"Hello\tWorld"`     |

function PalendramChecker() {
  const input = prompt("Enter Your Value !");
  let palendram = true;
  if (!input) {
    console.log("Invlide Input !");
    return;
  } else {
    let i = 0,
      j = input.length - 1;
    while (i < j) {
      if (input.charAt(i) !== input.charAt(j)) {
        palendram = false;
        break;
      }
    }
    palendram = true;
  }

  // condition to check the palendram value.

  if (palendram) console.log("This is a Palendram.")
  else console.log("This is not Palendram.")
}

PalendramChecker();
