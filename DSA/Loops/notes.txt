let number = Number(prompt("Enter Your Number!"));
if (isNaN(number)) {
  console.log("invalide Number !");
} else {
  if (number > 0) {
    let rev = 0;
    while (number > 0) {
      let rem = number % 10;
      // this is the equation for that it reverse the given value in prompt
      rev = rev * 10 + rem;
      number = Math.floor(number / 10);
    }
    console.log(rev);
  } else {
    console.log("Must be positive Greater than 1");
  }
}

let number = Number(prompt("Enter Your Number!"));

if (isNaN(number)) {
  console.log("Invalid Number!");
} else {
  if (number > 0) {
    let sum = 0; // move outside loop
    let temp = number; // store original value

    while (temp > 0) {
      let rem = temp % 10;
      let fact = 1;

      // Correct factorial loop
      for (let i = 1; i <= rem; i++) {
        fact *= i;
      }

      sum += fact;
      temp = Math.floor(temp / 10);
    }

    console.log("Sum of factorial of digits:", sum);
  } else {
    console.log("Must be positive and greater than 1");
  }
}
